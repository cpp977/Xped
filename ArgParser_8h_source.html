75ArgParser::
76ArgParser (int argc_input, char** argv_input)
77{
78 std::vector<string> args(argv_input+1, argv_input+argc_input);
79
80 for (int i=0; i<args.size(); ++i)
81 {
82 if (args[i].substr(0,1) == "-")
83 {
84 args[i].erase(0,1);
85
86 std::size_t found_eq = args[i].find("=");
87 // form "-arg=x"
88 if (found_eq != std::string::npos)
89 {
90 string repl_eq = args[i].replace(found_eq, 1, " ");
91 std::istringstream iss(repl_eq);
92 std::vector<string> split_eq{std::istream_iterator<string>{iss}, std::istream_iterator<string>{}};
93 assert(split_eq.size() == 2);
94 dictionary.insert(std::pair<string,string>(split_eq[0],split_eq[1]));
95 }
96 // form "-arg x" or "-arg"
97 else
98 {
99 if (i!=args.size()-1) // in the middle of arglist
100 {
101 // form "-arg x"
102 if (args[i+1].substr(0,1) != "-")
103 {
104 dictionary.insert(std::pair<string,string>(args[i],args[i+1])); // store -arg val
105 }
106 // form "-arg" is interpreted as "arg=1"
107 else
108 {
109 dictionary.insert(std::pair<string,string>(args[i],"1")); // store -arg 1 (true)
110 }
111 }
112 else // at the end of arglist
113 {
114 dictionary.insert(std::pair<string,string>(args[i],"1")); // store -arg 1 (true)
