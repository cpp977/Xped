130void Geometry2D::
131fill_HoppingMatrix (size_t range)
132{
133 assert(range>0 and range<=coupling_neighbor.size());
134
135 if (lattice_.size(0)==1) {assert(path != SNAKE and "Must use Lx>=2 with the SNAKE geometry!");}
136
137 HoppingMatrix[range-1].resize(lattice_.volume()*lattice_.unitCell.size(),lattice_.volume()*lattice_.unitCell.size()); HoppingMatrix[range-1].setZero();
138 if (coupling_neighbor[range-1] < 1.e-8) {return;}
139
140 // Mirrors the y coordinate to create a snake.
141 auto mirror = [this] (int iy) -> int
142 {
143 vector<int> v(lattice_.size(1));
144 std::iota (begin(v),end(v),0);
145 reverse(v.begin(),v.end());
146 return v[iy];
147 };
148
149 for (int ix=0; ix<lattice_.size(0); ++ix)
150 for (int iy=0; iy<lattice_.size(1); ++iy)
151 for (int jx=0; jx<lattice_.size(0); ++jx)
152 for (int jy=0; jy<lattice_.size(1); ++jy)
153 {
154 int countCellAtoms_i=-1;
155 for (const auto &[atom_i,position_i] : lattice_.unitCell)
156 {
157 countCellAtoms_i++;
158 int countCellAtoms_j=-1;
159 for (const auto &[atom_j,position_j] : lattice_.unitCell)
160 {
161 countCellAtoms_j++;
162 int iy_=iy, jy_=jy;
163 if (path == SNAKE)
164 {
165 // mirror even y only
166 iy_ = (ix%2==0)? iy : mirror(iy);
167 jy_ = (jx%2==0)? jy : mirror(jy);
168 }
169 // the index is calculated normally:
170 int index_i = iy*lattice_.unitCell.size()+lattice_.size(1)*lattice_.unitCell.size()*ix+countCellAtoms_i;
171 int index_j = jy*lattice_.unitCell.size()+lattice_.size(1)*lattice_.unitCell.size()*jx+countCellAtoms_j;
172
173 // but is stored together with the mirrored y_:
174 if (jx == 0 and jy == 0)
175 {
176 index[make_tuple(ix,iy_,atom_i)] = index_i;
177 coord[index_i] = make_tuple(ix,iy_,atom_i);
178// cout << "ix=" << ix << ", iy_=" << iy_ << ", index_i=" << index_i << endl;
179 }
180
181 if (lattice_.ARE_NEIGHBORS( {ix,iy_},{jx,jy_}, atom_i, atom_j, range))
182 {
183 HoppingMatrix[range-1](index_i,index_j) += coupling_neighbor[range-1];
184 number_of_bonds++;
185 }
186 }
187 }
188 }
189 assert(number_of_bonds%2 == 0);
190}
