113inline void HDF5Interface::
114switch_to (FILE_ACCESS_MODE mode_input)
115{
116 MODE = mode_input;
117 if (MODE == WRITE) { file = std::make_unique<H5::H5File>(filename.c_str(), H5F_ACC_TRUNC); }
118 else if (MODE == READ) { file = std::make_unique<H5::H5File>(filename.c_str(), H5F_ACC_RDONLY); }
119 else if (MODE == REWRITE) { file = std::make_unique<H5::H5File>(filename.c_str(), H5F_ACC_RDWR); }
120}
