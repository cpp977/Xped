16inline double uniformGrid (int index, double xmin, double xmax, int xpoints)
17{
18 if (xpoints == 1)
19 {
20 assert(abs(xmin-xmax)<numeric_limits<double>::epsilon());
21 return xmin;
22 }
23 else
24 {
25 double step = (xmax-xmin)/(xpoints-1);
26 return xmin + index*step;
27 }
28}
