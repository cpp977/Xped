89Lattice2D(array<size_t,2> L_input, array<bool,2> PERIODIC_input, LatticeType type_input, size_t furthest_neighbor)
90:L(L_input), PERIODIC(PERIODIC_input), type_(type_input)
91{
92 assert(furthest_neighbor >=1);
93 origin.setZero();
94 neighbor_distance.resize(furthest_neighbor);
95 if (type_ == LatticeType::SQUARE)
96 {
97 name_ = "Square lattice";
98 a[0] << 1. , 0.;
99 a[1] << 0. , 1.;
100 neighbor_distance[0] = a[0].norm();
101 if (furthest_neighbor == 2 and L[1] > 1) {neighbor_distance[1] = (a[0]+a[1]).norm();}
102 else if (furthest_neighbor == 2 and L[1] == 1) {neighbor_distance[1] = (2*a[0]).norm();}
103 unitCell[""]=Eigen::Matrix<double,dim,1>::Zero();
104 }
105 else if (type_ == LatticeType::TRIANG_XC or type_ == LatticeType::TRIANG)
106 {
107 if (type_ == LatticeType::TRIANG_XC) {name_ = "Triangular lattice (XC)";}
108 if (type_ == LatticeType::TRIANG) {name_ = "Triangular lattice";}
109 a[0] << 1. , 0.;
110 a[1] << 0.5 , std::sqrt(3.)*0.5 ;
111 neighbor_distance[0] = a[0].norm();
112 if (furthest_neighbor == 2 and L[1] > 1) {neighbor_distance[1] = (a[0]+a[1]).norm();}
113 else if (furthest_neighbor == 2 and L[1] == 1) {neighbor_distance[1] = (2*a[0]).norm();}
114 unitCell[""]=Eigen::Matrix<double,dim,1>::Zero();
115 }
116 else if (type_ == LatticeType::TRIANG_YC)
117 {
118 name_ = "Triangular lattice (YC)";
119 a[0] << std::sqrt(3.)*0.5 , 0.5 ;
120 a[1] << 0. , 1.;
121 neighbor_distance[0] = a[0].norm();
122 if (furthest_neighbor == 2 and L[1] > 1) {neighbor_distance[1] = (a[0]+a[1]).norm();}
123 else if (furthest_neighbor == 2 and L[1] == 1) {neighbor_distance[1] = (2*a[0]).norm();}
124 unitCell[""]=Eigen::Matrix<double,dim,1>::Zero();

126 Eigen::Rotation2D<double> rot90(0.5*M_PI);
127 Eigen::Matrix<double,dim,dim> rotate90 = rot90.toRotationMatrix();
128 b[0] = 2*M_PI * rotate90 * a[1] / a[0].dot(rotate90*a[1]);
129 b[1] = 2*M_PI * rotate90 * a[0] / a[1].dot(rotate90*a[0]);
130
131 assert(abs(a[0].dot(b[0])-2*M_PI) <1.e-12);
132 assert(abs(a[1].dot(b[1])-2*M_PI) <1.e-12);
133 assert(abs(a[0].dot(b[1])) <1.e-12);

137Eigen::Matrix<double,2,1> Lattice2D::
138getSite(const IndexSite& i) const
139{
140 Eigen::Matrix<double,2,1> out;
141 if (type_ == LatticeType::TRIANG or type_ == LatticeType::SQUARE)
142 {
143 out = i[0]*a[0] + i[1]*a[1];
144 }
145 else if (type_ == LatticeType::TRIANG_XC)
146 {
147 out = (i[0]-std::floor(static_cast<double>(i[1])/2.))*a[0] + i[1]*a[1];
148 }
149 else if (type_ == LatticeType::TRIANG_YC)
150 {
151 out = i[0]*a[0] + (i[1]-std::floor(static_cast<double>(i[0])/2.))*a[1];
152 }
153 else {assert(false and "Unknown LatticeType.");}
154 return out;
155}
