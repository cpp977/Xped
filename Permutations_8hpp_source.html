89 {
90 N = pi.size();
91 pi_inv.resize(N);
92 for(std::size_t i = 0; i < N; i++) { pi_inv[pi[i]] = i; }

96 auto start_it = std::find(visited.cbegin(), visited.cend(), false);
97 std::size_t start = std::distance(visited.cbegin(), start_it);
