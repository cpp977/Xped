18 double phi (const int &n, const double &h, const double &a, const double &b)
19 {
20 double t = n*h;
21
22 // careful limit t->0
23 return (n==0)? 1./(a+b+2.):
24 t/(1.-exp(-2.*t-a*(1.-exp(-t))-b*(exp(t)-1.)));
25 }

28 double phi_prime (const int &n, const double &h, const double &a, const double &b)
29 {
30 double t = n*h;
31
32 double denom = (1.-exp(-2.*t-a*(1.-exp(-t))-b*(exp(t)-1.)));
33 double double_exp = exp(-2.*t-a*(1.-exp(-t))-b*(exp(t)-1.));
34 double double_exp_deriv = -2.-a*exp(-t)-b*exp(t);
35
36 return (n==0)? 0.5*(a*a+2.*a*b+5.*a+b*b+3.*b+4.)/pow(a+b+2.,2):
37 t*double_exp*double_exp_deriv/pow(denom,2)+1./denom;
38 }
